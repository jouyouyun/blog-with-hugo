最近在处理路径时，要求统一用 =URI= (通用资源标识符) 的格式来表示。偶尔会遇到 URI 被编码的情况，这就需要解码了，故在此把 =Encode/Decode URI= 的方法记录在此，以备后用。

#+BEGIN_SRC go
package utils

import (
	"net/url"
)

/**
 * Input: filepath
 * Output: encode string
 */
func encodeURI (filepath string) string {
	if len(filepath) < 1 {
		return ""
	}

	u := url.URL{}
	u.Path = filepath
	return u.String()
}

/**
 * Input: URI string
 * Output: filepath
 */
func decodeURI (uri string) string {
	if len(uri) < 1 {
		return ""
	}

	u, err := url.Parse(uri)
	if err != nil {
		return ""
	}

	return u.Path
}
#+END_SRC
